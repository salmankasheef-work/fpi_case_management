📊 Google Sheets Integration
To sync case data to Google Sheets, deploy the Apps Script below and paste your URL.
STEP 1 — OPEN APPS SCRIPT
Go to script.google.com → New Project → paste the code below
STEP 2 — APPS SCRIPT CODE
function doPost(e) {
  const data = JSON.parse(e.postData.contents);
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  
  // Cases sheet
  let sheet = ss.getSheetByName("FPI_Cases") 
    || ss.insertSheet("FPI_Cases");
  
  if(sheet.getLastRow()===0){
    sheet.appendRow(["Case ID","Category","Fraud Type","Hub","City",
      "State","Value","Status","Priority","FPI POC","POD",
      "Police Action","Recovery","UTR","Trend","Terminated",
      "Date Filed","Synced At"]);
  }
  
  // Find existing row or append
  const vals = sheet.getDataRange().getValues();
  let rowIdx = -1;
  for(let i=1;i<vals.length;i++){
    if(vals[i][0]===data.case_id){ rowIdx=i+1; break; }
  }
  
  const row = [data.case_id,data.category,data.fraud_type,
    data.hub,data.city,data.state,data.value,data.status,
    data.priority,data.poc_name,data.poc_pod,data.police_action,
    data.recovery,data.utr,data.trend,data.riders_terminated,
    data.date_filed,new Date()];
  
  if(rowIdx>0) sheet.getRange(rowIdx,1,1,row.length).setValues([row]);
  else sheet.appendRow(row);
  
  return ContentService.createTextOutput("OK");
}
STEP 3 — DEPLOY & PASTE URL
Deploy as: Web App → Execute as: Me → Who has access: Anyone
